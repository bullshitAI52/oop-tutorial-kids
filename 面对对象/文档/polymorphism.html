多态示例

Python 代码

class Animal:
    def speak(self):
        raise NotImplementedError

class Dog(Animal):
    def speak(self):
        return "汪汪"

class Cat(Animal):
    def speak(self):
        return "喵喵"

def make_speak(a: Animal):
    print(a.speak())

if __name__ == "__main__":
    make_speak(Dog())
    make_speak(Cat())
