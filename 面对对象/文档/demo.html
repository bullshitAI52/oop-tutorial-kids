class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

class Employee(Person):
    def __init__(self, name, age, role):
        super().__init__(name, age)
        self.role = role

    def describe(self):
        return f"{self.name} 是 {self.role}，{self.age} 岁"

if __name__ == "__main__":
    p = Person("小美", 28)
    e = Employee("小美", 28, "开发者")
    print(p.name, p.age)
    print(e.describe())
