封装示例

Python 代码

class Person:
    def __init__(self, name):
        self._name = name  # 约定的保护属性

    @property
    def name(self):
        return self._name

    @name.setter
    def name(self, value):
        self._name = value

class SecretHolder:
    def __init__(self, secret):
        self.__secret = secret  # 私有属性（名字改写，外部不可直接访问）

    @property
    def secret(self):
        return self.__secret

    @secret.setter
    def secret(self, value):
        self.__secret = value

if __name__ == "__main__":
    p = Person("小李")
    print(p.name)
    s = SecretHolder("隐藏信息")
    print(s.secret)
